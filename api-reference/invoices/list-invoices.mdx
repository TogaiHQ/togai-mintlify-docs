<ParamField query="owner_id" type="string">
  Unique identifier of the owner corresponding to the invoice[account or an invoice group]
</ParamField>
<ParamField query="customer_id" type="string">
  Unique identifier of the customer corresponding to the invoice
</ParamField>
<ParamField query="price_plan_id" type="string">
  Unique identifier of the price plan corresponding to the invoice
</ParamField>
<ParamField query="status" type="enum">
  Status of the invoice (`DRAFT`, `DUE`, `PAID`, `VOID`, `UN_COLLECTIBLE`, `INITIATE_REFUND`, `REFUND_COMPLETED`, `MERGED`)
</ParamField>
<ParamField query="class" type="enum">
  Class to which the invoice is categorized (`INVOICE`, `ORDER`, `BILLABLE`)
</ParamField>
<ParamField query="start_date" type="date_time">
  Start date of the invoicing period
</ParamField>
<ParamField query="end_date" type="date_time">
  End date of the invoicing period
</ParamField>
</Accordion>
